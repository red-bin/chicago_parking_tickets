    def parse_addr(addr):
    parsed = usaddress.parse(addr)
    addr_dict = dict([ (key, val) for val, key in parsed ])

    streetname = addr_dict['StreetName']
    if 'StreetNamePreDirectional' in addr_dict:
        streetdir = addr_dict['StreetNamePreDirectional']
    else:
        streetdir = ''
        addr_dict['StreetNamePreDirectional'] = ''

    if streetname not in unique_mapstreets:
        fixed_street = correct_street(streetname, streetdir)
        if fixed_street:
            addr_dict['StreetName'] = fixed_street
        if not fixed_street:
            return None

    ret_addr = pd.Series(addr_dict)

    return ret_addr
